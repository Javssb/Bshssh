<script src="https://cdn.jsdelivr.net/npm/three@0.152.2/build/three.min.js"></script>

<script>

const scene = new THREE.Scene();
scene.background = new THREE.Color(0xd2b48c);

const camera = new THREE.PerspectiveCamera(
75,
window.innerWidth/window.innerHeight,
0.1,
1000
);

const renderer = new THREE.WebGLRenderer({
antialias:true
});

renderer.setSize(
window.innerWidth,
window.innerHeight
);

document.body.appendChild(renderer.domElement);

const light = new THREE.DirectionalLight(0xffffff,1);

light.position.set(5,10,5);

scene.add(light);

const ground = new THREE.Mesh(

new THREE.PlaneGeometry(300,300),

new THREE.MeshStandardMaterial({
color:0xc2b280
})

);

ground.rotation.x = -Math.PI/2;

scene.add(ground);

const player = new THREE.Mesh(

new THREE.BoxGeometry(1,2,1),

new THREE.MeshStandardMaterial({
color:0x00ff00
})

);

player.position.y = 1;

scene.add(player);

camera.position.set(0,5,8);

let velocityY = 0;
let gravity = -0.01;
let kills = 0;

const enemies=[];

for(let i=0;i<5;i++){

const enemy = new THREE.Mesh(

new THREE.BoxGeometry(1,2,1),

new THREE.MeshStandardMaterial({
color:0xff0000
})

);

enemy.position.set(
Math.random()*20-10,
1,
Math.random()*-20
);

scene.add(enemy);

enemies.push(enemy);

}

document.getElementById("jump").onclick=()=>{

if(player.position.y<=1.01){

velocityY = 0.25;

}

};

document.getElementById("shoot").onclick=()=>{

const bullet = new THREE.Mesh(

new THREE.SphereGeometry(0.2),

new THREE.MeshBasicMaterial({
color:0xffff00
})

);

bullet.position.copy(player.position);

scene.add(bullet);

function animateBullet(){

bullet.position.z -=0.5;

enemies.forEach((enemy,index)=>{

if(enemy && bullet.position.distanceTo(enemy.position)<1){

scene.remove(enemy);

enemies[index]=null;

kills++;

document.getElementById("hud").innerHTML=
`
LAND ARAB 🔥<br>
HP: 100<br>
Kills: ${kills}
`;

}

});

requestAnimationFrame(animateBullet);

}

animateBullet();

};

const joystick = document.getElementById("joystick");

let moveX=0;
let moveZ=0;

joystick.addEventListener("touchmove",(e)=>{

const touch = e.touches[0];

const rect = joystick.getBoundingClientRect();

const x = touch.clientX - rect.left - 60;
const y = touch.clientY - rect.top - 60;

moveX = x*0.01;
moveZ = y*0.01;

});

joystick.addEventListener("touchend",()=>{

moveX=0;
moveZ=0;

});

function animate(){

requestAnimationFrame(animate);

player.position.x += moveX;

player.position.z += moveZ;

velocityY += gravity;

player.position.y += velocityY;

if(player.position.y<1){

player.position.y=1;

velocityY=0;

}

camera.position.x = player.position.x;

camera.position.z = player.position.z + 8;

camera.lookAt(player.position);

renderer.render(scene,camera);

}

animate();

window.addEventListener("resize",()=>{

camera.aspect=
window.innerWidth/window.innerHeight;

camera.updateProjectionMatrix();

renderer.setSize(
window.innerWidth,
window.innerHeight
);

});
